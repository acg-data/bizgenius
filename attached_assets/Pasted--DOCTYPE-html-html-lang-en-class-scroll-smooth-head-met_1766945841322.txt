    <script>
        lucide.createIcons();
        AOS.init({ duration: 800, once: true });

        // --- 1. Live Ticker Logic ---
        const tickerEl = document.getElementById('tickerText');
        const tickerContainer = document.getElementById('liveTicker');
        const events = [
            "Someone in Austin generated a 'Taco Truck' Business Plan",
            "Someone in London created a SaaS Pitch Deck",
            "A user in NYC just exported Financial Projections",
            "Someone in Toronto launched a 'Pet Hotel' website"
        ];
        
        let eventIndex = 0;

        function updateTicker() {
            // Fade out
            tickerContainer.classList.remove('opacity-100', 'translate-y-0');
            tickerContainer.classList.add('opacity-0', 'translate-y-4');

            setTimeout(() => {
                tickerEl.innerText = events[eventIndex];
                eventIndex = (eventIndex + 1) % events.length;
                
                // Fade in
                tickerContainer.classList.remove('opacity-0', 'translate-y-4');
                tickerContainer.classList.add('opacity-100', 'translate-y-0');
            }, 500);
        }

        // Start Ticker
        setTimeout(() => {
            tickerContainer.classList.remove('opacity-0', 'translate-y-4');
            tickerContainer.classList.add('opacity-100', 'translate-y-0');
            setInterval(updateTicker, 5000);
        }, 1000);


        // --- 2. Typewriter Logic ---
        const phrases = [
            "A subscription coffee service...", 
            "A SaaS for dog walkers...", 
            "A boutique hotel in Tulum...", 
            "An AI legal consultant..."
        ];
        const typewriterEl = document.getElementById('typewriter');
        const realInput = document.getElementById('realInput');
        let phraseIndex = 0;
        let charIndex = 0;
        let isDeleting = false;
        let isUserTyping = false;

        function type() {
            if (isUserTyping) return; // Stop animation if user types

            const currentPhrase = phrases[phraseIndex];
            
            if (isDeleting) {
                typewriterEl.innerText = currentPhrase.substring(0, charIndex - 1);
                charIndex--;
            } else {
                typewriterEl.innerText = currentPhrase.substring(0, charIndex + 1);
                charIndex++;
            }

            let typeSpeed = isDeleting ? 30 : 80;

            if (!isDeleting && charIndex === currentPhrase.length) {
                typeSpeed = 2000; // Pause at end
                isDeleting = true;
            } else if (isDeleting && charIndex === 0) {
                isDeleting = false;
                phraseIndex = (phraseIndex + 1) % phrases.length;
                typeSpeed = 500;
            }

            setTimeout(type, typeSpeed);
        }

        function handleTyping(input) {
            isUserTyping = true;
            typewriterEl.style.display = 'none'; // Hide typewriter text
            document.querySelector('.cursor').style.display = 'none';
        }

        // Start Typewriter
        setTimeout(type, 1000);

    </script>
